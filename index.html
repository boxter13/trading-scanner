function analyzeQuoteOnly(symbol, q, item, market){ const a = { signal: 'NO BUY', score: 0, reasons: [], entry: null, stop: null, target: null, qty: 0, riskValue: 0 }; if(!q || !q.c || !q.o || !q.h || !q.l || !q.pc){ a.reasons.push('Données incomplètes'); return a; } const current = q.c; const open = q.o; const high = q.h; const low = q.l; const moveFromOpen = ((current - open) / open) * 100; const range = Math.max(0.01, high - low); const closeNearHigh = (high - current) / range < 0.25; const strongMove = moveFromOpen > 0.3; const notTooLate = moveFromOpen < 1.5; const aboveOpen = current > open; const zl = parseNum(item.zoneLow); const zh = parseNum(item.zoneHigh); let inZone = true; if(zl != null && zh != null){ inZone = current >= zl && current <= zh; } // 🔥 CONDITIONS SNIPER const sniperBuy = market.allowBuy && aboveOpen && closeNearHigh && strongMove && notTooLate && inZone; if(sniperBuy){ a.signal = 'BUY'; a.score = 10; a.reasons.push('SNIPER VALIDÉ'); } else { a.signal = 'NO BUY'; } // Gestion position a.entry = Math.round(current * 100) / 100; a.stop = Math.round(current * (1 - (parseNum(item.stopPct) ?? 0.8)/100) * 100) / 100; a.target = Math.round(current * (1 + (parseNum(item.targetPct) ?? 1.5)/100) * 100) / 100; const risk = APP.cfg.capital * (APP.cfg.riskPct / 100); const perShare = Math.abs(a.entry - a.stop); a.qty = Math.floor(risk / perShare); return a; }
